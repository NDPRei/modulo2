CREATE TABLE Formação 
( 
 Curso VARCHAR(n),  
 Data Inicio INT,  
 Descrição VARCHAR(n),  
 idPerfil INT,  
 Data FIm INT,  
); 

CREATE TABLE Experiência 
( 
 Nome da Empresa VARCHAR(n) PRIMARY KEY,  
 Data Inicio INT,  
 Cargo VARCHAR(n),  
 idPerfil INT,  
 Data Fim INT,  
); 

CREATE TABLE Realizações 
( 
 Descrição VARCHAR(n),  
 Data INT,  
 idPerfil INT,  
); 

CREATE TABLE Habilidades 
( 
 Habilidades INT,  
 idPerfil INT,  
); 

CREATE TABLE Personalidade 
( 
 Descrição VARCHAR(n),  
 idPerfil INT,  
); 

CREATE TABLE Perfil 
( 
 Foto VARCHAR(n),  
 Nome VARCHAR(n),  
 Cargo INT,  
 Endereço VARCHAR(n),  
 Email VARCHAR(n),  
 Contato INT,  
 ID INT PRIMARY KEY AUTO_INCREMENT,  
); 

ALTER TABLE Formação ADD FOREIGN KEY(idPerfil) REFERENCES Perfil (idPerfil)
ALTER TABLE Experiência ADD FOREIGN KEY(idPerfil) REFERENCES Perfil (idPerfil)
ALTER TABLE Realizações ADD FOREIGN KEY(idPerfil) REFERENCES Perfil (idPerfil)
ALTER TABLE Habilidades ADD FOREIGN KEY(idPerfil) REFERENCES Perfil (idPerfil)
